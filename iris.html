import numpy as np
from sklearn.metrics import accuracy_score
from scipy.spatial.distance import minkowski


def uniform_weight_function(dist):
    return 1.0

def inverse_weight_function(dist):
    return 1. / (dist + 1E-6)


class KNN:
    def __init__(self, n_neighbors=5, weights='uniform', p=2):
        self.n_neighbors = n_neighbors
        if weights == 'uniform':
            self.weights = uniform_weight_function
        elif weights == 'distance':
            self.weights = inverse_weight_function
        else:
            raise NotImplementedError
        self.distance = lambda u, v: minkowski(u, v, p)

    def fit(self, X, y):
        self.X = np.array(X)
        self.y = np.array(y)
        if self.X.shape[0] != self.y.shape[0]:
            raise ValueError("Array dimensions are not consistent.")

    def predict(self, X):
        X = np.array(X)
        ans = []
        classes = set(self.y)
        for sample_idx, v in enumerate(X):
            predicts = {cls: 0.0 for cls in classes}
            score_list = [(self.distance(x, v), self.y[i]) for i, x in enumerate(self.X)]
            score_list.sort()
            scores = np.array([x[0] for x in score_list])
            for i in range(min(self.n_neighbors, len(score_list))):
                predicts[score_list[i][1]] += self.weights(score_list[i][0])
            ans.append(max([(x[1], x[0]) for x in predicts.items()])[1])
        return np.array(ans)

    def score(self, X, y):
        return accuracy_score(self.predict(X), y)
